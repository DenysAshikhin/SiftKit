func get_random_boss(num_bosses:int) -> Array:

			var pulled_boss = possible_bosses.pick_random();
			if(pulled_boss == BOSS_IDS.scanilus):
				possible_bosses = [BOSS_IDS.necrotilus, BOSS_IDS.bastionlus];
			bosses_to_spawn_final.push_back(pulled_boss);

		return bosses_to_spawn_final;

		Global._throw_error("Invalid number of bosses specified");

	return boss_ids;

# Spawns each requested boss id with a short stagger while rejecting malformed payloads.
# Why:
# - Empty/invalid arrays are authoritative caller bugs and should be surfaced through `Global._throw_error(...)`.
# - Early rejection avoids no-op spawn calls that can desync jump-boss gate state from actual boss presence.
func _spawn_bosses(bosses_arr:Array) -> void:
	if(bosses_arr.size() == 0):
		Global._throw_error("boss spawn request rejected: empty boss array");
		return;
	if(!_are_requested_boss_ids_valid(bosses_arr)):
		Global._throw_error("boss spawn request rejected: invalid boss id in array " + str(bosses_arr));
		return;
	for boss in bosses_arr:
		if(Global.RESETTING):
			return;
		_spawn_boss({"boss":boss});
		await Global._sleep(0.25);

# Returns true when every requested boss id maps to a known boss prefab key.
# Why:
# - Jump-boss orchestration can receive malformed arrays from external callers/debug tooling.
# - Pre-validating avoids entering active jump-boss gate state when the payload cannot spawn.
func _are_requested_boss_ids_valid(bosses_arr:Array) -> bool:
	var valid_ids:Array = BOSS_IDS.values();
	for boss_id in bosses_arr:
		if(str(boss_id) not in valid_ids):
			return false;
	return true;

func _return_bullet(bullet):

	#if(Global.Game_Over or Global.JUMPING):
		#if(!bullet.destroy_queued):
			#USED_BULLETS.erase(bullet.ID);
			#bullet.queue_free();
		#return;

	if(bullet.destroy_queued):
		return;
	else:
		bullet.destroy_queued = true;

	# Anti-Waste Volley (targetting_2 option 3): 50% chance to re-fire on miss
	if (bullet.elapsed >= bullet.time_to_live and bullet.get('extra_params') != null and !bullet.extra_params.get("refire", false)):
		var source = bullet.source_unit;
		var did_refire:bool = false;
		# Guard teardown-time stale references: pooled bullets can outlive their source unit
		# for a frame during harness shutdown.
		if (source and source.custom_class == "drone" and "targetting_2" in source.tech_unlocks):
			var t2_sel = source.tech_unlocks["targetting_2"].options.current_selection;
			if (t2_sel == "3"):
				var t2_cfg = source.tech_unlocks["targetting_2"].options.options["3"];
				if (Global.randomf() <= t2_cfg.get('refire_chance', 0.50)):
					var new_params = bullet.extra_params.duplicate();
					new_params["refire"] = true;
					new_params["origin"] = source.global_position;
					if (is_instance_valid(source.target)):
						new_params["target"] = source.target.global_position;
						call_deferred("_spawn_bullet", source, bullet.DAMAGE, new_params);
						did_refire = true;

		# Waste Reduction (target_lead_1 option 3): chance to re-fire on miss
		if (!did_refire and source and source.custom_class == "drone" and "target_lead_1" in source.tech_unlocks):
			var tl1_sel = source.tech_unlocks["target_lead_1"].options.current_selection;
			if (tl1_sel == "3"):
				var tl1_cfg = source.tech_unlocks["target_lead_1"].options.options["3"];
				if (Global.randomf() <= tl1_cfg.get('refire_chance', 0.30)):
					var new_params = bullet.extra_params.duplicate();
					new_params["refire"] = true;
					new_params["origin"] = source.global_position;
					if(new_params.has("target")):
						new_params.erase("target");
					call_deferred("_spawn_bullet", source, bullet.DAMAGE, new_params);

	bullet.set_deferred("visible", false);
	#bullet.visible = false;
	bullet.set_deferred("process_mode", Node.PROCESS_MODE_DISABLED);
	#bullet.process_mode = Node.PROCESS_MODE_DISABLED;
	USED_BULLETS.erase(bullet.ID);
	BULLETS_PUSHBACK.push_back(bullet);
	#await Global.await_deferred();
	#AVAILABLE_BULLETS.push_back(bullet);

		temp_enemy.CUSTOM_DEFAULT_STATS = { };
		temp_enemy._setup(ID_INNER, TEAM);

		temp_enemy.set_deferred("process_mode", Node.PROCESS_MODE_DISABLED);
		AVAILABLE_SUPPORTS.push_back(temp_enemy);
		return;

	temp_enemy._reset_for_respawn();

# Spawns one boss instance using either runtime/default stats or caller-provided custom stats.
# Why:
# - Explicit boss id validation prevents null-instance follow-on failures when malformed payloads arrive.
# - Tutorial-chain checks remain authoritative to guarantee the required boss appears in guided flows.
func _spawn_boss(params, updateCounter = true):
	randomize();
	if(typeof(params) != TYPE_DICTIONARY or !params.has("boss")):
		Global._throw_error("boss spawn request rejected: missing `boss` id in params");
		return;
	var boss_id:String = str(params.boss);
	var enforce_tutorial_chain_gate: bool = true;
	if(typeof(params) == TYPE_DICTIONARY and bool(params.get("ignore_tutorial_chain_gate", false))):
		enforce_tutorial_chain_gate = false;

	var required_boss := Unlocks.tutorial_chain_required_boss_id();
	if(enforce_tutorial_chain_gate and required_boss != "" and boss_id != required_boss):
		Global._throw_error("Tutorial-chain boss spawn mismatch. Required: " + str(required_boss) + " Got: " + boss_id);
		return;

	var temp_enemy;
	if (Global.is_sector()):
		sector_boss_count += 1;

	Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.boss_encountered, 1);
	match Global.Map_Manager.get_zone_title():
		"Easy Zone":
			Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.boss_encountered_easy, 1);
		"Medium Zone":
			Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.boss_encountered_medium, 1);
		"Hard Zone":
			Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.boss_encountered_hard, 1);

	match(boss_id):
		BOSS_IDS.necrotilus:
			temp_enemy = necrotilus.instantiate();
			Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.necrotilus_encountered, 1);
			match Global.Map_Manager.get_zone_title():
				"Easy Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.necrotilus_encountered_easy, 1);
				"Medium Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.necrotilus_encountered_medium, 1);
				"Hard Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.necrotilus_encountered_hard, 1);
		BOSS_IDS.scanilus:
			temp_enemy = scanilus.instantiate();
			Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.scanilus_encountered, 1);
			match Global.Map_Manager.get_zone_title():
				"Easy Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.scanilus_encountered_easy, 1);
				"Medium Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.scanilus_encountered_medium, 1);
				"Hard Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.scanilus_encountered_hard, 1);
		BOSS_IDS.bastionlus:
			temp_enemy = bastionlus.instantiate();
			Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.bastionlus_encountered, 1);
			match Global.Map_Manager.get_zone_title():
				"Easy Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.bastionlus_encountered_easy, 1);
				"Medium Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.bastionlus_encountered_medium, 1);
				"Hard Zone":
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.bastionlus_encountered_hard, 1);
		_:
			Global._throw_error("boss spawn request rejected: unknown boss id `" + boss_id + "`");
			return;


	temp_enemy.position = Vector2(
		randf_range(Global.play_capital_trans.origin.x - 350, Global.play_capital_trans.origin.x + 350),
			randf_range(Global.play_capital_trans.origin.y - 350, Global.play_capital_trans.origin.y + 350)
			)

	if(params.has('CUSTOM_STATS')):
		temp_enemy.CUSTOM_STATS = params['CUSTOM_STATS'];
		temp_enemy.CUSTOM_DEFAULT_STATS = params['CUSTOM_DEFAULT_STATS'];
		if(params.has("tech_unlocks")):
			temp_enemy.tech_unlocks = params["tech_unlocks"];
		if(params.has("items")):
			temp_enemy.tech_unlocks = params["items"];
		enemies[params['CUSTOM_STATS']['id']] = temp_enemy
		bosses[params['CUSTOM_STATS']['id']] = temp_enemy;
		temp_enemy._setup(params['CUSTOM_STATS']['id'], TEAM)
	else:
		if(updateCounter):
			bossCounter += 1;
			Map_Manager._update_enemy_counters(1);
		temp_enemy._setup(Map_Manager.enemyCounter, TEAM)
		enemies[Map_Manager.enemyCounter] = temp_enemy
		bosses[Map_Manager.enemyCounter] = temp_enemy;
	add_child(temp_enemy)


func _map_created_unit(unit) -> void:
	match unit.custom_class:
		"drone":
			drones[unit.ID] = unit;
		"support":
			supports[unit.ID] = unit;
		"frigate":
			frigates[unit.ID] = unit;

	enemies[unit.ID] = unit;

var sector_boss_failed_rolls:float = 0.0;
var sector_boss_grace:float = 0.0;#this is the 15 seconds after killing boss that gets set to allow jumps
var enemy_death_elapse:float = 0.0;#how much time elapsed since last enemy died
var enemy_death_elapse_start:float = 10.0;#what we want to start ticking down from, each time it runs out, we tick this down to smaller (to a min)

var min_enemy_y : float = -1;
var max_enemy_y : float = -1;
var min_enemy_x : float = -1;
var max_enemy_x : float = -1;
const min_alive_ticks:int = 10;
var avg_enemy_position = Vector2(0, 0);

func _physics_process(delta:float) -> void:
	if(Global.TUTORIAL.tutorial_combat_paused()):
		return;


	avg_enemy_position = Vector2(0, 0);
	max_enemy_x = -1;
	max_enemy_y = -1;
	min_enemy_x = -1;
	min_enemy_y = -1;
	var count:float = 0.0;

	for key in frigates:
		var orig = frigates[key].INNER_TRANSFORM.origin;
		if(frigates[key].lived_ticks > min_alive_ticks):
			avg_enemy_position += orig;
			count += 1;
			if(max_enemy_x == -1 || orig.x > max_enemy_x):
				max_enemy_x = orig.x;
			if(max_enemy_y == -1 || orig.y > max_enemy_y):
				max_enemy_y = orig.y;
			if(min_enemy_x == -1 || orig.x < min_enemy_x):
				min_enemy_x = orig.x;
			if(min_enemy_y == -1 || orig.y < min_enemy_y):
				min_enemy_y = orig.y;
		frigates[key]._core_logic_tick(delta);
	for key in drones:
		var orig = drones[key].INNER_TRANSFORM.origin;
		if(drones[key].lived_ticks > min_alive_ticks):
			avg_enemy_position += orig;
			count += 1;
			if(max_enemy_x == -1 || orig.x > max_enemy_x):
				max_enemy_x = orig.x;
			if(max_enemy_y == -1 || orig.y > max_enemy_y):
				max_enemy_y = orig.y;
			if(min_enemy_x == -1 || orig.x < min_enemy_x):
				min_enemy_x = orig.x;
			if(min_enemy_y == -1 || orig.y < min_enemy_y):
				min_enemy_y = orig.y;
		drones[key]._core_logic_tick(delta);
	for key in supports:
		var orig = supports[key].INNER_TRANSFORM.origin;
		if(supports[key].lived_ticks > min_alive_ticks):
			avg_enemy_position += orig;
			count += 1;
			if(max_enemy_x == -1 || orig.x > max_enemy_x):
				max_enemy_x = orig.x;
			if(max_enemy_y == -1 || orig.y > max_enemy_y):
				max_enemy_y = orig.y;
			if(min_enemy_x == -1 || orig.x < min_enemy_x):
				min_enemy_x = orig.x;
			if(min_enemy_y == -1 || orig.y < min_enemy_y):
				min_enemy_y = orig.y;
		supports[key]._core_logic_tick(delta);
	for key in bosses:
		avg_enemy_position += bosses[key].INNER_TRANSFORM.origin;
		count += 1;
		if(max_enemy_x == -1 || bosses[key].INNER_TRANSFORM.origin.x > max_enemy_x):
			max_enemy_x = bosses[key].INNER_TRANSFORM.origin.x;
		if(max_enemy_y == -1 || bosses[key].INNER_TRANSFORM.origin.y > max_enemy_y):
			max_enemy_y = bosses[key].INNER_TRANSFORM.origin.y;
		if(min_enemy_x == -1 || bosses[key].INNER_TRANSFORM.origin.x < min_enemy_x):
			min_enemy_x = bosses[key].INNER_TRANSFORM.origin.x;
		if(min_enemy_y == -1 || bosses[key].INNER_TRANSFORM.origin.y < min_enemy_y):
			min_enemy_y = bosses[key].INNER_TRANSFORM.origin.y;
		bosses[key]._core_logic_tick(delta);

	if(count > 0.0):
		avg_enemy_position /= count;
	sector_boss_grace -= delta;
	if(!initialised):
		return;

	if(Global.Game_Over):
		return;


	if(Global.JUMPING and !list_clearing):
		sector_boss_elapsed = 0.0;
		list_clearing = true;
		Global._queue_list(DRONES_PUSHBACK);
		Global._queue_list(SUPPORTS_PUSHBACK);
		Global._queue_list(FRIGATES_PUSHBACK);
		#Global._queue_list(BULLETS_PUSHBACK);
	if(!Global.JUMPING and list_clearing):
		list_clearing = false;

	while (BULLETS_PUSHBACK.size() > 0):
		AVAILABLE_BULLETS.push_back(BULLETS_PUSHBACK.pop_back());

	while (DRONES_PUSHBACK.size() > 0):
		var drone_temp = DRONES_PUSHBACK.pop_back();
		match drone_temp.TIER:
			1:
				AVAILABLE_DRONES.push_back(drone_temp);
			2:
				AVAILABLE_DRONES_2.push_back(drone_temp);
			3:
				AVAILABLE_DRONES_3.push_back(drone_temp);
			4:
				AVAILABLE_DRONES_4.push_back(drone_temp);

			6:
				AVAILABLE_DRONES_6.push_back(drone_temp);
			7:
				AVAILABLE_DRONES_7.push_back(drone_temp);
			8:
				AVAILABLE_DRONES_8.push_back(drone_temp);
			9:
				AVAILABLE_DRONES_9.push_back(drone_temp);
			10:
				AVAILABLE_DRONES_10.push_back(drone_temp);

	while (SUPPORTS_PUSHBACK.size() > 0):
		var next_sup = SUPPORTS_PUSHBACK.pop_back();
		AVAILABLE_SUPPORTS.push_back(next_sup);

	while (FRIGATES_PUSHBACK.size() > 0):
		var frigate_temp = FRIGATES_PUSHBACK.pop_back();
		match frigate_temp.TIER:
			1:
				AVAILABLE_FRIGATES.push_back(frigate_temp);
			2:
				AVAILABLE_FRIGATES_2.push_back(frigate_temp);
			3:
				AVAILABLE_FRIGATES_3.push_back(frigate_temp);
			4:
				AVAILABLE_FRIGATES_4.push_back(frigate_temp);
			5:
				AVAILABLE_FRIGATES_5.push_back(frigate_temp);
			6:
				AVAILABLE_FRIGATES_6.push_back(frigate_temp);
			7:
				AVAILABLE_FRIGATES_7.push_back(frigate_temp);
			8:
				AVAILABLE_FRIGATES_8.push_back(frigate_temp);
			9:
				AVAILABLE_FRIGATES_9.push_back(frigate_temp);
			10:
				AVAILABLE_FRIGATES_10.push_back(frigate_temp);


	drone_loop = true;
	frigate_loop = true;
	enemy_death_elapse += delta;



	if(enemy_death_elapse >= enemy_death_elapse_start and !Global._is_home_base()):
		enemy_death_elapse = 0.0;
		enemy_death_elapse_start = max(0.2, enemy_death_elapse_start - 0.5);
		Map_Manager._update_enemy_counters(1);





	if (initialised == false):
		return;
	if (Map_Manager.initialised == false):
		return;
	var can_spawn:bool = Global.can_spawn_units();
	if(can_spawn):
		if(spawn_jump_boss):
			spawn_jump_boss = false;
			if(jump_bosses.size() == 0):
				Global._throw_error("jump bosses not pre-seeded?");
				spawn_jump_boss_active = false;
				jump_bosses = [];
			elif(!_are_requested_boss_ids_valid(jump_bosses)):

				spawn_jump_boss_active = false;
				jump_bosses = [];
			else:
				var bosses_before_spawn:int = bosses.size();
				_spawn_bosses(jump_bosses);
				jump_bosses = [];
				spawn_jump_boss_active = bosses.size() > bosses_before_spawn;
				if(!spawn_jump_boss_active):
					Global._throw_error("jump boss spawn request did not create any active boss");
		elif(spawn_jump_boss_active):
			pass;
		elif (should_spawn_boss):
			should_spawn_boss = false;
			_spawn_bosses(get_random_boss(bosses_to_spawn));
			bosses_to_spawn = 1;
		elif(Global.is_sector()):
			sector_boss_elapsed += delta;
			var base_chance:float = 0.15 * (1.0 + sector_boss_failed_rolls);#10% chance to spawn
			#var chance_time:float = max(15.0 - sector_boss_count - sector_boss_failed_rolls, 5.0);#10 seconds for each boss rollspawn
			var chance_time:float = max(15.0 - sector_boss_count, 7.5);#10 seconds for each boss roll

			#var max_num_bosses:int = 1 + max(Global.Map_Manager.CURR_ZONE_MODIFIERS.sector_modifiers.corruptions - 2, 0);

					#max_num_bosses += 1;
				#"Hard Zone":
					#max_num_bosses += 2;

			if(sector_boss_elapsed >= chance_time):
				sector_boss_elapsed = 0.0;
				if(Global.randomf() < base_chance and bosses.size() < max_num_bosses):
					_spawn_bosses(get_random_boss(1));
					sector_boss_failed_rolls = 0;
				else:
					sector_boss_failed_rolls += 1;


	if (
		spawn_jump_boss_active
		or (!Global.is_sector() and bosses.size() > 0)
		):
		return;

	var scanilus_found:bool = false;
	for boss_key in bosses:
		#early terminate, we don't want scanilus twice

	for i in range(10):
		frigate_limit.push_back(temp_frigate_limit % tier_two_frigate);

			max_frigate_limit = i + 1;

func generate_alloy_string() -> String:
	var stringy:String = "";
	var choices:Array = [];
	var selected_zone_title:String = Global.Map_Manager.CURR_ZONE_MODIFIERS.title;
	match(selected_zone_title):
		"Easy Zone":
			choices = ["1", "1-2", "1-3", "2", "2-3", "3"];
		"Medium Zone":
			choices = ["2", "2-3", "2-4", "3", "3-4", "4"];
		"Hard Zone":
			choices = ["3", "3-4", "3-5", "4", "4-5", "5"];

	stringy = Global.math.pick_random_elements(choices, 1)[0];
	return stringy;

func _delete_enemy(unit_to_die, _other_params:={}):

	enemy_death_elapse = 0.0;

	if(unit_to_die.queued_into_pool):
		return;
	else:
		unit_to_die.queued_into_pool = true;



	var value = unit_to_die.DEFAULT_STATS.value;
	value *= Global.Map_Manager.CURR_ZONE_MODIFIERS['value_multiplier'];
	value *= Global.BUFFS.active_buffs[Global.BUFFS.possible_buffs.value_boost];
	value *= Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.enemy_value];

	var drolium_mult:float = 1.0;
	drolium_mult *= Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.enemy_drolium];
	var buf_equip = Player.buff_equipped("value_boost_equip");
	if(buf_equip.equipped and buf_equip.buff.active):
		var value_buff_speed_boost = Unlocks.get_tech("mothership_unlocks", "item_value_boost_equip");
		if(value_buff_speed_boost.unlocked):
			if(value_buff_speed_boost.options.current_selection == "1"):
				drolium_mult = value_buff_speed_boost.options.options["1"].drollium_gain;

	var rp_mult:float = 1.0;
	rp_mult *= Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.enemy_rp];
	var tech_buff = Unlocks.get_tech("mothership_unlocks", "feature_permanent_research");
	if(tech_buff.unlocked and tech_buff.options.current_selection != "-1"):
		if ("slag" in tech_buff.options.options[tech_buff.options.current_selection]):
			drolium_mult *=  tech_buff.options.options[tech_buff.options.current_selection].slag;
		if ("rp" in tech_buff.options.options[tech_buff.options.current_selection]):
			rp_mult *=  tech_buff.options.options[tech_buff.options.current_selection].rp;

	if(Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.enemy_alloy_on_kill] > 1.0):
		var chance = Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.enemy_alloy_on_kill] - 1.0;
		if(Global.randomf() < chance):
			Player.alloy_inventory[generate_alloy_string()].amount += 0.01 * value;
			Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.enemy_alloy_on_kill, 0.01 * value);

	if(Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.rp_progress_on_kill] > 1.0):
		var chance = Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.rp_progress_on_kill] - 1.0;
		if(Global.randomf() < chance):
			var active_researches:Array = [];
			for research in Player.current_research:
				if(research.time_remaining > 0 && research.status == Player.research_status.working):
					active_researches.push_back(research);
			if(active_researches.size() > 0):
				var chosen = active_researches.pick_random();
				Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.rp_progress_on_kill, chosen.time_remaining * 0.01);
				chosen.time_remaining *= 0.99;
				#chosen.time_remaining -= chosen.original_time * 0.01;
				#Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.rp_progress_on_kill, chosen.original_time * 0.01);

	match(unit_to_die.custom_class):
		'drone':
			drones.erase(unit_to_die.ID);
			Player._update_slag(value * drolium_mult, {"combat": true});
			Player._update_research_points(value * rp_mult, {"combat": true});
			Player.score += value;
			match unit_to_die.TIER:
				1:
					USED_DRONES.erase(unit_to_die.ID);
				2:
					USED_DRONES_2.erase(unit_to_die.ID);

				4:

				5:
					USED_DRONES_5.erase(unit_to_die.ID);
				6:
					USED_DRONES_6.erase(unit_to_die.ID);
				7:
					USED_DRONES_7.erase(unit_to_die.ID);
				8:
					USED_DRONES_8.erase(unit_to_die.ID);
				9:
					USED_DRONES_9.erase(unit_to_die.ID);
				10:
					USED_DRONES_10.erase(unit_to_die.ID);

		'support':

			unit_to_die._reset_linked_allies();
			supports.erase(unit_to_die.ID);
			Player._update_slag(value * drolium_mult, {"combat": true});
			Player._update_research_points(value * rp_mult, {"combat": true});
			Player.score += value;
			USED_SUPPORTS.erase(unit_to_die.ID);
		'frigate':
			frigates.erase(unit_to_die.ID);
			Player._update_slag(value * drolium_mult, {"combat": true});
			Player._update_research_points(value * rp_mult, {"combat": true});
			Player.score += value;
			match unit_to_die.TIER:
				1:
					USED_FRIGATES.erase(unit_to_die.ID);
				2:
					USED_FRIGATES_2.erase(unit_to_die.ID);
				3:
					USED_FRIGATES_3.erase(unit_to_die.ID);
				4:
					USED_FRIGATES_4.erase(unit_to_die.ID);
				5:
					USED_FRIGATES_5.erase(unit_to_die.ID);
				6:
					USED_FRIGATES_6.erase(unit_to_die.ID);
				7:
					USED_FRIGATES_7.erase(unit_to_die.ID);
				8:
					USED_FRIGATES_8.erase(unit_to_die.ID);
				9:
					USED_FRIGATES_9.erase(unit_to_die.ID);
				10:
					USED_FRIGATES_10.erase(unit_to_die.ID);
		'boss':
			bosses.erase(unit_to_die.ID);
			Player._update_slag(value * drolium_mult, {"combat": true});
			Player._update_research_points(value * rp_mult, {"combat": true});
			Player.score += value;
			sector_boss_grace = 15.0;

			if (Global.PRESTIGE.prestige_count > 0):
				var spawn_core:bool = false;
				if (Global.is_sector()):
					spawn_core = 0.01 >= Global.randomf();
				else:
					spawn_core = true;
				if (spawn_core):
					match Global.Map_Manager.get_zone_title():
						"Easy Zone":
							Unlocks.increase_rqmt_amount("easy_core", 1);
						"Medium Zone":
							Unlocks.increase_rqmt_amount("medium_core", 1);
						"Hard Zone":
							Unlocks.increase_rqmt_amount("hard_core", 1);

			if(spawn_jump_boss_active):
				spawn_jump_boss_active = false;


	Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.research_points_total, {"value": value * rp_mult, "display_order": 1}, {"is_combat": true, });
	Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.drolium_total, value * drolium_mult);
	enemies.erase(unit_to_die.ID);

	if(unit_to_die.custom_class == "boss"):
		unit_to_die.queue_free();
		return;

	unit_to_die._death_cleanup();
	unit_to_die.set_deferred("visible", false);
	unit_to_die.set_deferred("process_mode", Node.PROCESS_MODE_DISABLED);


	match(unit_to_die.custom_class):
		'drone':
			DRONES_PUSHBACK.push_back(unit_to_die);
		'support':
			SUPPORTS_PUSHBACK.push_back(unit_to_die);
		'frigate':
			FRIGATES_PUSHBACK.push_back(unit_to_die);

# Full enemy resets must silently prune dead registry entries so later physics ticks do not
# keep touching half-destroyed enemy state from the same transition.
func _prune_stale_enemy_reset_ref(registry, key, unit_class_name):
	registry.erase(key);
	var unit_id:int = int(key);
	enemies.erase(unit_id);
	match unit_class_name:
		"drone":
			_erase_used_drone_id(unit_id);
		"support":
			USED_SUPPORTS.erase(unit_id);
		"frigate":
			_erase_used_frigate_id(unit_id);

# Clears a drone id from every tier bucket because stale refs can no longer provide trustworthy tier data.

# Full enemy resets should drain every pooled bullet reference instead of trimming half the pool.
# That keeps teardown deterministic for tests and avoids leaked bullet nodes/resources surviving exit.
func _queue_and_clear_enemy_bullets() -> void:
	var queued_bullets:Array = [];
	queued_bullets.append_array(BULLETS_PUSHBACK);
	queued_bullets.append_array(AVAILABLE_BULLETS);
	for bullet_key in USED_BULLETS.keys():
		queued_bullets.push_back(USED_BULLETS[bullet_key]);
	Global._queue_list(queued_bullets);
	BULLETS_PUSHBACK.clear();
	AVAILABLE_BULLETS.clear();
	USED_BULLETS.clear();

func generate_save_data():
	var save_data = {
		"drones": {},
		"droneCounter": droneCounter,
		"supports": {},
		"supportCounter": supportCounter,
		"frigates": {},
		"frigateCounter": frigateCounter,
		"bosses": {},
		"bossCounter": bossCounter,
		"tech_stealing": tech_stealing,
		"num_bosses_spawned": num_bosses_spawned,
		"jump_bosses": jump_bosses,
		"sector_boss_count": sector_boss_count,
	}
	for key in drones.keys():
		save_data['drones'][key] = Global.generate_entity_save_data(drones[key]);
	for key in supports.keys():
		save_data['supports'][key] = Global.generate_entity_save_data(supports[key]);
	for key in frigates.keys():
		save_data['frigates'][key] = Global.generate_entity_save_data(frigates[key]);
	for key in bosses.keys():
		save_data['bosses'][key] = Global.generate_entity_save_data(bosses[key]);
	return save_data;

func _load_save_data(save_data):

	#return;
	await Global.saver._save_file_loaded();
	while(!is_instance_valid(Global.player_capital)):
		if(Global.RESETTING):
			return;
		await Global._sleep(0.1);

	if(Global.can_spawn_units()):

		if("drones" in save_data):
			for key in save_data['drones'].keys():
				_spawn_drone(Global.get_entity_save_data(save_data['drones'][key]), false);
		self.droneCounter = save_data['droneCounter'];

		if("supports" in save_data):
			for key in save_data['supports'].keys():
				_spawn_support(Global.get_entity_save_data(save_data['supports'][key]), false);
		self.supportCounter = save_data['supportCounter'];

		if("frigates" in save_data):
			for key in save_data['frigates'].keys():
				_spawn_frigate(Global.get_entity_save_data(save_data['frigates'][key]), false);
		self.frigateCounter = save_data['frigateCounter'];


		if("bosses" in save_data):
			for key in save_data['bosses'].keys():
				var boss_params = Global.get_entity_save_data(save_data['bosses'][key]);
				boss_params.boss = save_data['bosses'][key].CUSTOM_DEFAULT_STATS.sub_class;
				_spawn_boss(boss_params, false);
		self.bossCounter = save_data['bossCounter'];

		if("tech_stealing" in save_data):
			self.tech_stealing.populated = save_data['tech_stealing'].populated;
			for unit_type in save_data['tech_stealing'].available:
				for tech in save_data['tech_stealing'].available[unit_type]:
					match unit_type:
						"drone":
							tech = Unlocks.get_tech("drone_unlocks", tech.id);
						"support":
							tech = Unlocks.get_tech("support_unlocks", tech.id);
						"frigate":
							tech = Unlocks.get_tech("frigate_unlocks", tech.id);
			for unit_type in save_data['tech_stealing'].stolen:
				for tech in save_data['tech_stealing'].stolen[unit_type]:
					match unit_type:
						"drone":
							tech = Unlocks.get_tech("drone_unlocks", tech);
						"support":
							tech = Unlocks.get_tech("support_unlocks", tech);
						"frigate":
							tech = Unlocks.get_tech("frigate_unlocks", tech);

		self.tech_stealing = save_data['tech_stealing'];

	if ("num_bosses_spawned" in save_data):
		self.num_bosses_spawned = save_data['num_bosses_spawned'];
	if ("jump_bosses" in save_data):
		self.jump_bosses = save_data['jump_bosses'];
	if ("sector_boss_count" in save_data):
		self.sector_boss_count = save_data['sector_boss_count'];

	data_loaded = true;
