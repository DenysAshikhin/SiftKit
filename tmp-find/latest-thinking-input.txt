# Selects one random loading tip each time this scene is entered.
# - Tip source is markdown (`summary/loading_screen_tips.md`) parsed at runtime.
# - Any file/parsing failure fails loud via `Global._throw_error(...)`.
# - We still render a deterministic fallback in non-editor builds.
func _select_random_loading_tip() -> void:
	loading_tips = _load_loading_tips_from_markdown();
	if(loading_tips.is_empty()):
		Global._throw_error(
			"loading_screen._select_random_loading_tip: no parseable tips found. path="
			+ LOADING_TIPS_PATH
			+ " open_error="
			+ str(FileAccess.get_open_error())
		);
		current_tip_text = FALLBACK_LOADING_TIP;
	else:
		current_tip_text = str(loading_tips.pick_random());

	TIP_LABEL.text = "Tip: " + current_tip_text;

# Loads and parses numbered markdown tip lines (`<int>. <text>`) into plain text.
# - Heading/separator/non-numbered lines are intentionally ignored.
# - Inline bold markers are stripped so UI renders plain text.
func _load_loading_tips_from_markdown() -> Array[String]:
	var markdown_text:String = FileAccess.get_file_as_string(LOADING_TIPS_PATH);
	if(markdown_text == ""):
		Global._throw_error(
			"loading_screen._load_loading_tips_from_markdown failed to read tips file. path="
			+ LOADING_TIPS_PATH
			+ " open_error="
			+ str(FileAccess.get_open_error())
		);
		return [];

	var parsed_tips:Array[String] = [];
	var lines:PackedStringArray = markdown_text.split("\n", false);
	for raw_line:String in lines:
		var tip_text:String = _extract_tip_from_markdown_line(raw_line);
		if(tip_text != ""):
			parsed_tips.push_back(tip_text);
	return parsed_tips;

# Extracts one tip string from a markdown list line.
# Expected shape:
# - `<number>. <tip text>`
# Returns empty string when the line is not a tip entry.
func _extract_tip_from_markdown_line(raw_line:String) -> String:
	var trimmed_line:String = raw_line.strip_edges();
	if(trimmed_line == ""):
		return "";

	var split_line:PackedStringArray = trimmed_line.split(". ", false, 1);
	if(split_line.size() != 2):
		return "";
	if(!split_line[0].is_valid_int()):
		return "";

	var cleaned_tip:String = split_line[1].strip_edges();
	if(cleaned_tip == ""):
		return "";
	cleaned_tip = cleaned_tip.replace("**", "");
	return cleaned_tip;

		temp_enemy.items = params["items"];

	temp_enemy._setup(id_to_use, 1);
	# Fail-loud setup contracts (e.g. malformed persisted `STATE`) reject this instance before runtime mapping.
	if(bool(temp_enemy.get("_spawn_rejected"))):
		if(is_fresh_spawn):
			active_supports = max(0, active_supports - 1);
		temp_enemy.set_deferred("visible", false);

		AVAILABLE_SUPPORTS.push_back(temp_enemy);

var UNIT_CAP_INCREASE:int = 0;

func getDroneLimit() -> int:
	var buf_equip = Player.buff_equipped("production_drones_equip");
	UNIT_CAP_INCREASE = 0;
	if(buf_equip.equipped and buf_equip.buff.active):
		var prod_buff_speed_boost = Unlocks.get_tech("mothership_unlocks", "item_production_drones_equip");
		if(prod_buff_speed_boost.unlocked):
			if(prod_buff_speed_boost.options.current_selection == "1"):
				UNIT_CAP_INCREASE = prod_buff_speed_boost.options.options["1"].cap_boost;
	var constants_mult:float = Constants.GLOBAL_STATS[Constants.GLOBAL_STAT_KEY.drone_limit_mult];
	var limit:float = (Motherships.CURR_Drone_Limit + UNIT_CAP_INCREASE + DRONE_CAP_INCREASE) * constants_mult;
	return roundi(limit);

func getDroneCount() -> int:
	var drone_count:int = 0;
	for i in range(1, active_drone_arr.size()-1):
		drone_count += active_drone_arr[i] * pow(5, i-1);
	return drone_count;
func getFrigateCount() -> int:
	var frigate_count:int = 0;
	for i in range(1, active_frigate_arr.size()-1):
		frigate_count += active_frigate_arr[i] * pow(5, i-1);
	return frigate_count;

# Total cap is an integer sum of per-unit integer limits.
func getMaxShip() -> int:
	return getDroneLimit() + getSupportLimit() + getFrigateLimit();
func getActiveShip() -> int:
	return getDroneCount() + active_supports + getFrigateCount();

func getUnusedShip() -> int:
	var droneLimit = getDroneLimit();
	var supportLimit = getSupportLimit();
	var frigateLimit = getFrigateLimit();

	var missing:int = (
		(droneLimit - getDroneCount())
		+(supportLimit - active_supports)
		+(frigateLimit - getFrigateCount())
		)

	return missing;

func _physics_process(delta:float):
	#print("friendly units:" + str(units.size()) + " ------ "
	#+ "friendly bullets: " + str(USED_BULLETS.size()));

	if(Global.TUTORIAL.tutorial_combat_paused()):
		return;

	var count:int = 0;
	avg_friendly_position = Vector2(0, 0);
	max_friendly_x = -1;
	max_friendly_y = -1;
	min_friendly_x = -1;
	min_friendly_y = -1;

	var necro_drones:Array = [[],[],[],[],[],[],[],[],[],[]];
	var necro_frigates:Array = [[],[],[],[],[],[],[],[],[],[]];


	for key in frigates:
		#USED_FRIGATES[key].call_deferred("_core_logic_tick", delta);
		var unit = frigates[key];
		var orig = unit.INNER_TRANSFORM.origin;

		if(unit.DEFAULT_STATS.is_zombie):
			necro_frigates[unit.TIER-1].push_back(unit);
			if (necro_frigates[unit.TIER-1].size() > Global.Enemy_Manager.tier_two_frigate):
				call_deferred("merge_units", necro_frigates[unit.TIER-1]);
				necro_frigates[unit.TIER-1] = [];

		if(unit.lived_ticks > min_alive_ticks):
			avg_friendly_position += orig;
			count += 1;
			if(max_friendly_x == -1 || orig.x > max_friendly_x):
				max_friendly_x = orig.x;
			if(max_friendly_y == -1 || orig.y > max_friendly_y):
				max_friendly_y = orig.y;
			if(min_friendly_x == -1 || orig.x < min_friendly_x):
				min_friendly_x = orig.x;
			if(min_friendly_y == -1 || orig.y < min_friendly_y):
				min_friendly_y = orig.y;
		unit._core_logic_tick(delta);

	for key in drones:
		#unit.call_deferred("_core_logic_tick", delta);
		var unit = drones[key];
		var orig = unit.INNER_TRANSFORM.origin;

		if (unit.DEFAULT_STATS.is_zombie):
			#t1_necro_drones.push_back(key);
			necro_drones[unit.TIER-1].push_back(unit);
			if (necro_drones[unit.TIER-1].size() > Global.Enemy_Manager.tier_two_drone):
				call_deferred("merge_units", necro_drones[unit.TIER-1]);
				necro_drones[unit.TIER-1] = [];

		if(unit.lived_ticks > min_alive_ticks):
			avg_friendly_position += orig;
			count += 1;
			if(max_friendly_x == -1 || orig.x > max_friendly_x):
				max_friendly_x = orig.x;
			if(max_friendly_y == -1 || orig.y > max_friendly_y):
				max_friendly_y = orig.y;
			if(min_friendly_x == -1 || orig.x < min_friendly_x):
				min_friendly_x = orig.x;
			if(min_friendly_y == -1 || orig.y < min_friendly_y):
				min_friendly_y = orig.y;
		unit._core_logic_tick(delta);

	for key in USED_SUPPORTS:
		#USED_SUPPORTS[key].call_deferred("_core_logic_tick", delta);
		var orig = USED_SUPPORTS[key].INNER_TRANSFORM.origin;

		if(USED_SUPPORTS[key].lived_ticks > min_alive_ticks):
			avg_friendly_position += orig;
			count += 1;
			if(max_friendly_x == -1 || orig.x > max_friendly_x):
				max_friendly_x = orig.x;
			if(max_friendly_y == -1 || orig.y > max_friendly_y):
				max_friendly_y = orig.y;
			if(min_friendly_x == -1 || orig.x < min_friendly_x):
				min_friendly_x = orig.x;
			if(min_friendly_y == -1 || orig.y < min_friendly_y):
				min_friendly_y = orig.y;
		USED_SUPPORTS[key]._core_logic_tick(delta);


	for key in life_rafts:
		life_rafts[key]._core_logic_tick(delta);

	if(count > 0):
		avg_friendly_position /= count;

	if(Global.saver.need_to_load_from_save):
		return;

	if(Global.Game_Over):
		return;





	if(Global.JUMPING and !list_clearing):
		list_clearing = true;
		Global._queue_list(DRONES_PUSHBACK);
		Global._queue_list(SUPPORTS_PUSHBACK);
		Global._queue_list(FRIGATES_PUSHBACK);


	if(!Global.JUMPING and list_clearing):
		list_clearing = false;

	while (BULLETS_PUSHBACK.size() > 0):
		var temp = BULLETS_PUSHBACK.pop_back();
		AVAILABLE_BULLETS.push_back(temp);

	while (DRONES_PUSHBACK.size() > 0):
		var drone_temp = DRONES_PUSHBACK.pop_back();
		match drone_temp.TIER:
			1:
				AVAILABLE_DRONES.push_back(drone_temp);
			2:
				AVAILABLE_DRONES_2.push_back(drone_temp);

				AVAILABLE_DRONES_3.push_back(drone_temp);
			4:
				AVAILABLE_DRONES_4.push_back(drone_temp);
			5:
				AVAILABLE_DRONES_5.push_back(drone_temp);
			6:
				AVAILABLE_DRONES_6.push_back(drone_temp);
			7:
				AVAILABLE_DRONES_7.push_back(drone_temp);
			8:
				AVAILABLE_DRONES_8.push_back(drone_temp);
			9:
				AVAILABLE_DRONES_9.push_back(drone_temp);
			10:
				AVAILABLE_DRONES_10.push_back(drone_temp);

	while (SUPPORTS_PUSHBACK.size() > 0):
		var next_sup = SUPPORTS_PUSHBACK.pop_back();
		AVAILABLE_SUPPORTS.push_back(next_sup);

	while (FRIGATES_PUSHBACK.size() > 0):
		var frigate_temp = FRIGATES_PUSHBACK.pop_back();
		match frigate_temp.TIER:
			1:
				AVAILABLE_FRIGATES.push_back(frigate_temp);
			2:
				AVAILABLE_FRIGATES_2.push_back(frigate_temp);
			3:
				AVAILABLE_FRIGATES_3.push_back(frigate_temp);
			4:
				AVAILABLE_FRIGATES_4.push_back(frigate_temp);
			5:
				AVAILABLE_FRIGATES_5.push_back(frigate_temp);
			6:
				AVAILABLE_FRIGATES_6.push_back(frigate_temp);
			7:
				AVAILABLE_FRIGATES_7.push_back(frigate_temp);
			8:
				AVAILABLE_FRIGATES_8.push_back(frigate_temp);
			9:
				AVAILABLE_FRIGATES_9.push_back(frigate_temp);
			10:
				AVAILABLE_FRIGATES_10.push_back(frigate_temp);


	drone_loop = true;
	frigate_loop = true;



	var production_elapsed = delta * Global.BUFFS.active_buffs[Global.BUFFS.possible_buffs.production_boost];
	var active_ratio:float = float(getActiveShip()) / float(max(1, getMaxShip()));
	var prod_boost_missing = max(1.0, 1.0 + (Global.BASES.upgrades.mothership_prod_boost_2.bonus * (1.0 - active_ratio)));
	production_elapsed *= prod_boost_missing;

	var tech_buff = Unlocks.get_tech("mothership_unlocks", "feature_permanent_research");
	if(tech_buff.unlocked and tech_buff.options.current_selection != "-1"):
		if ("unit_speed" in tech_buff.options.options[tech_buff.options.current_selection]):
			production_elapsed *=  tech_buff.options.options[tech_buff.options.current_selection].unit_speed;


	if(Global.JUMPING):
		return;

	if(Global._is_home_base()):
		droneSpawnElapsed = 0;
		droneSpawnElapsedMax = 0;
		supportSpawnElapsed = 0;
		supportSpawnElapsedMax = 0;
		frigateSpawnElapsed = 0;
		frigateSpawnElapsedMax = 0;

	var temp_drone_limit:int = getDroneLimit();
	var drone_limit:Array = [];
	for i in range(10):
		drone_limit.push_back(temp_drone_limit % tier_two_drone);
		temp_drone_limit = floori(float(temp_drone_limit) / float(tier_two_drone));

			#-2 is fresh jump in, check what we need and spawn it, then move on as normal
			if(next_drone_tier == -2):

				if(temp_tier != -1 and Unlocks.get_tech("mothership_unlocks", "drones_unlock").unlocked):
					call_deferred("_spawn_drone", {"tier": temp_tier});
					Unlocks._update_objective_map(Unlocks.OBJECT_KEYS.drone_num_created, {"value": temp_tier}, {"is_combat": true});

			elif (next_drone_tier != -1):

		elif (active_drone_arr[1] < getDroneLimit()):

## Initializes canonical unit identity (`ID`, `TEAM`) and enforces team contract with fail-loud behavior.
## Why:
## - Existing entity `_setup(id, team)` implementations duplicated team validation/error text.
## - A single helper ensures every unit reports malformed setup consistently and deterministically.
## Behavior:
## - Invalid `team` triggers `Global._throw_error(...)` and falls back to `fallback_team`.
## - Invalid `fallback_team` is also reported and forced to team `2`.
func _setup_unit_identity(id, team, owner_name:String, fallback_team:int = 2) -> void:
	ID = int(id);
	TEAM = int(team);
	if(TEAM == 1 or TEAM == 2):
		return;

	Global._throw_error(owner_name + "._setup invalid team=" + str(team) + " for id=" + str(ID));
	if(fallback_team != 1 and fallback_team != 2):
		Global._throw_error(owner_name + "._setup invalid fallback_team=" + str(fallback_team) + " for id=" + str(ID) + ". forcing fallback team=2");
		fallback_team = 2;
	TEAM = fallback_team;

## Applies one shared `CUSTOM_STATS` key/value pair using the canonical transform/identity contract.
## Why:
## - Drone/Frigate/Support duplicated the same `id/global_position/global_rotation/else` branch logic.
## - Centralizing this keeps pooled spawn payload handling consistent across actor classes.
## Behavior:
## - `id` is always ignored (identity comes from `_setup(...)` / `_setup_unit_identity(...)`).
## - `global_position` and `global_rotation` mutate `INNER_TRANSFORM` deterministically and defer `rotation` sync.
## - All other keys are assigned to `self[key]`.
func _apply_custom_stat_key(key:String, value) -> void:
	if(key == "id"):
		return;
	if(key == "global_position"):
		INNER_TRANSFORM.origin = Vector2(value.x, value.y);
		return;
	if(key == "global_rotation"):
		var orig = INNER_TRANSFORM.origin;
		INNER_TRANSFORM = INNER_TRANSFORM.rotated(value);
		INNER_TRANSFORM.origin = orig;
		set_deferred("rotation", INNER_TRANSFORM.get_rotation());
		return;
	self[key] = value;
